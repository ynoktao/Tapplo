1    package com.sazemTapplo
2    
3    import android.content.Intent
4    import android.os.Bundle
5    import android.widget.Toast
6    import androidx.appcompat.app.AppCompatActivity
7    import com.sazemTapplo.databinding.ActivityMainBinding
8    
9    class MainActivity : AppCompatActivity() {
10   
11       private lateinit var binding: ActivityMainBinding
12   
13       override fun onCreate(savedInstanceState: Bundle?) {
14           super.onCreate(savedInstanceState)
15           binding = ActivityMainBinding.inflate(layoutInflater)
16           setContentView(binding.root)
17   
18           binding.createTableButton.setOnClickListener {
19               val columnCountText = binding.columnCountEditText.text.toString()
20               if (columnCountText.isEmpty()) {
21                   Toast.makeText(this, "Lütfen sütun sayısını girin", Toast.LENGTH_SHORT).show()
22               } else {
23                   val columnCount = columnCountText.toInt()
24                   val intent = Intent(this, TableActivity::class.java)
25                   intent.putExtra("COLUMN_COUNT", columnCount)
26                   startActivity(intent)
27               }
28           }
29       }
30   }
